Python调用示例（支持所有OpenAI模型）


# OpenAI API - Python快速调用示例
import requests

API_KEY = "your_api_key"  # 您在平台获取的API密钥 
API_URL = "https://api.zzzzapi.com/v1/chat/completions"  # OpenAI API中转地址

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

data = {
    "model": "gpt-4o",  # 支持OpenAI全系列模型：gpt-3.5-turbo, gpt-4, gpt-4o等
    "messages": [
        {"role": "system", "content": "你是一个专业的AI助手"},
        {"role": "user", "content": "请帮我写一个Python爬虫程序"}
    ],
    "temperature": 0.7,
    "max_tokens": 2000
}

response = requests.post(API_URL, headers=headers, json=data)
print(response.json()["choices"][0]["message"]["content"])
